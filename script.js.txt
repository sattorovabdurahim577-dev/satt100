let users = { "005667407": { pass: "12", name: "Admin", isAdmin: true, avatar: "", isVerified: true } };
let currentUser = null;
let verificationRequests = [];
let replyData = null;

function toggleAuth(isReg) {
    document.getElementById('reg-fields').style.display = isReg ? 'block' : 'none';
    document.getElementById('login-fields').style.display = isReg ? 'none' : 'block';
}

function registerUser() {
    const ph = document.getElementById('reg-phone').value;
    const p1 = document.getElementById('reg-pass1').value;
    const p2 = document.getElementById('reg-pass2').value;
    if (!ph || !p1) return alert("Заполните поля");
    if (p1 !== p2) return alert("Пароли не совпадают");
    if (users[ph]) return alert("Номер занят");
    
    // Регистратсия ва мустақиман даромадан
    users[ph] = { pass: p1, name: "User_"+ph.slice(-4), isAdmin: false, avatar: "", isVerified: false };
    currentUser = { ...users[ph], phone: ph };
    updateUI(); showScreen('main-screen'); startSlider();
}

function loginUser() {
    const ph = document.getElementById('log-phone').value;
    const ps = document.getElementById('log-pass').value;
    if (users[ph] && users[ph].pass === ps) {
        currentUser = { ...users[ph], phone: ph };
        updateUI(); showScreen('main-screen'); startSlider();
    } else alert("Ошибка входа");
}

function requestVerification() {
    if (currentUser.isVerified) return alert("Уже верифицирован");
    if (verificationRequests.some(r => r.phone === currentUser.phone)) return alert("Заявка уже отправлена");
    verificationRequests.push({ phone: currentUser.phone, name: currentUser.name || currentUser.phone });
    alert("Заявка отправлена админу!");
}

function updateRequestsList() {
    const list = document.getElementById('requests-list');
    list.innerHTML = verificationRequests.length ? "" : "<p style='font-size:11px;color:#888'>Заявок нет</p>";
    verificationRequests.forEach((req, i) => {
        const div = document.createElement('div');
        div.className = "req-item";
        div.innerHTML = `<span>${req.name}</span><div>
            <button onclick="approveVerify(${i})" style="color:var(--green);background:none;border:none;cursor:pointer">✅</button>
            <button onclick="rejectVerify(${i})" style="color:red;background:none;border:none;cursor:pointer;margin-left:10px">❌</button>
        </div>`;
        list.appendChild(div);
    });
}

function approveVerify(i) {
    users[verificationRequests[i].phone].isVerified = true;
    if(currentUser.phone === verificationRequests[i].phone) currentUser.isVerified = true;
    verificationRequests.splice(i, 1); updateRequestsList(); updateUI();
}
function rejectVerify(i) { verificationRequests.splice(i, 1); updateRequestsList(); }

function sendMessage() {
    const inp = document.getElementById('msg-input');
    if (!inp.value.trim()) return;
    const box = document.getElementById('chat-messages');
    const wrap = document.createElement('div');
    wrap.className = "msg-wrapper my";
    const name = currentUser.name || currentUser.phone;
    const badge = currentUser.isVerified ? `<img src="https://upload.wikimedia.org/wikipedia/commons/e/e4/Twitter_Verified_Badge.svg" style="width:12px;margin-left:3px">` : "";
    
    wrap.innerHTML = `
        <div class="msg-avatar">${currentUser.avatar ? `<img src="${currentUser.avatar}">` : name[0]}</div>
        <div class="msg-content">
            <span class="msg-username">${name}${badge}</span>
            <div class="msg-body">
                ${replyData ? `<small style="display:block;opacity:0.6;border-bottom:1px solid #fff3;margin-bottom:4px">↩️ ${replyData}</small>` : ""}
                ${inp.value}
                <small class="atvet-btn" onclick="setReply('${inp.value}')">Ответить</small>
            </div>
        </div>`;
    box.appendChild(wrap); inp.value = ""; cancelReply(); box.scrollTop = box.scrollHeight;
}

function setReply(t) { replyData = t; document.getElementById('reply-box').style.display = 'flex'; document.querySelector('.reply-text').innerText = t; }
function cancelReply() { replyData = null; document.getElementById('reply-box').style.display = 'none'; }

    const name = currentUser.name || currentUser.phone;
    document.getElementById('my-name-display').innerText = name;

    if (currentUser.isAdmin) { document.getElementById('admin-requests-section').style.display = "block"; updateRequestsList(); }
}
